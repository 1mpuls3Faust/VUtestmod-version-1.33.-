country_event = { # Losing grip on northern coast part 1
	id = vu_fall_of_rome.1
	title = vu_fall_of_rome.1.t
	desc = vu_fall_of_rome.1.d
	picture = CONQUEST_eventPicture
	
	fire_only_once = yes
	
	goto = 1764
	
	trigger = {
		tag = TUR
		is_year = 1448
		exists = BYZ
		is_great_power = yes
		OR = {
			1764 = { owned_by = BYZ }
			7139 = { owned_by = BYZ }
			7138 = { owned_by = BYZ }
		}
		NOT = {
			BYZ = {
				has_won_war_against = {
					who = PREV
					max_years_since = 25
				}
			}
		}
	}
	
	mean_time_to_happen = {
		months = 3
	}
	
	option = { # We will make demands
		name = vu_fall_of_rome.1.a
		highlight = yes
		ai_chance = { factor = 100 } #AI will always make a demand
		BYZ = {
			country_event = { 
				id = vu_fall_of_rome.2
				days = 1
			}
		}
	}
	
	option = { # Pass
		name = vu_fall_of_rome.1.b
		ai_chance = { factor = 0 }
		add_prestige = -5
		add_mil_power = 10
	}
}

country_event = { # Losing grip on northern coast part 2
	id = vu_fall_of_rome.2
	title = vu_fall_of_rome.2.t
	desc = vu_fall_of_rome.2.d
	picture = BANKRUPTCY_eventPicture
	
	major = yes
	
	is_triggered_only = yes
	
	goto = 1764
	
	option = { # We will agree
		name = vu_fall_of_rome.2.a
		highlight = yes
		ai_chance = { factor = 40 }
		if = {
			limit = {
				1764 = { owned_by = ROOT }
			}
			1764 = {
				cede_province = TUR
				add_territorial_core = TUR
			}
		}
		if = {
			limit = {
				7139 = { owned_by = ROOT }
			}
			7139 = {
				cede_province = TUR
				add_territorial_core = TUR
			}
		}
		if = {
			limit = {
				7138 = { owned_by = ROOT }
			}
			7138 = {
				cede_province = TUR
				add_territorial_core = TUR
			}
		}
		add_dip_power = 25
		add_prestige = -20
		add_adm_power = 25
		add_stability = 1
	}
	
	option = { # No, we will fight
		name = vu_fall_of_rome.2.b
		highlight = yes
		ai_chance = { factor = 60 }
		1764 = { add_territorial_core = TUR }
		7139 = { add_territorial_core = TUR }
		7138 = { add_territorial_core = TUR }
		6931 = { add_permanent_claim = TUR }
		add_mil_power = 25
		add_legitimacy = 10
		add_dip_power = -20
		TUR = {
			add_casus_belli = {
				target = ROOT
				type = cb_insult
				months = 12
			}
		}
	}
}
